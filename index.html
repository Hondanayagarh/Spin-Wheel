function doGet(e) { var token = e.parameter.token || ""; if (!token) { return ContentService.createTextOutput( JSON.stringify({status:"error", message:"No token"}) ).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput( JSON.stringify({status:"ok", token: token, message:"Token received"}) ).setMimeType(ContentService.MimeType.JSON); }
